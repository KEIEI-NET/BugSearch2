    def __init__(self, max_concurrent: int = 10):

        self.completed = []

    def add_job(self, job_config: Dict) -> str:
        """ã‚¸ãƒ§ãƒ–è¿½åŠ ã¨å„ªå…ˆåº¦ã‚½ãƒ¼ãƒˆ"""
